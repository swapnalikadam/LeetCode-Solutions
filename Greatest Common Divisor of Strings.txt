function gcdOfStrings(str1: string, str2: string): string {
    if(!((str1+str2)==(str2+str1))){
        return "";
    }

    const a:number = str1.length;
    const b:number = str2.length;
    let gcd=0;

    if(a>b){
        gcd = GCD(a, b)
    }else{
        gcd = GCD(b, a)
    }
    return str1.substring(0, gcd)
};

function GCD(a:number, b:number):number{
    if(b==0){
        return a;
    }
    return GCD(b, a%b);
}
